# Installation
# NOTE THAT FIGLET IS OPTIONAL TO INSTALL
Linux Installation:
sudo apt install python3 python2 git figlet
pip install tqdm
git clone https://github.com/Ronthehero/Crunch-2.0.git

Termux installation:
apt install python3 python2 git figlet
pip install tqdm
git clone https://github.com/Ronthehero/Crunch-2.0.git

# How to run Crunch 2.0 (python3 method):

cd Crunch-2.0
python3 Crunch-2.0

# How to run Crunch 2.0 (python2 method):

cd Crunch-2.0
python2 Crunch-2.0
